import {NextFunction, Request, Response, Router} from "express";
import {ObjectId} from "mongodb";
import {Bill} from "../baseClasses/Bill";
import {Frequency} from "../baseClasses/Frequency";
import {Purchase} from "../baseClasses/Purchase";
import DbClient = require("../DbClient");
 
export class BillController extends Bill {
 
    public static create(router: Router) {
 
        // Form with bill-id, item-id (ex. ObjectId("xxxxxxxxxxxxxxxxxxxxxxxx"))
        router.post("/purchases/add", (req: Request, res: Response, next: NextFunction) => {
            new BillController(req.query.user).addPurchase(res, req);
        });
 
        // Form with bill-id, item-id (ex. ObjectId("xxxxxxxxxxxxxxxxxxxxxxxx"))
        router.post("/purchases/remove", (req: Request, res: Response, next: NextFunction) => {
            new BillController(req.body.user).removePurchase(res, req);
        });
 
        // Form with bill-id, item-id (ex. ObjectId("xxxxxxxxxxxxxxxxxxxxxxxx"))
        router.get("/purchases", (req: Request, res: Response, next: NextFunction) => {
            new BillController(req.query.user).getPurchases(res, req);
        });
 
        // Form with bill-id, item-id (ex. ObjectId("xxxxxxxxxxxxxxxxxxxxxxxx"))
        router.post("/bill/create", (req: Request, res: Response, next: NextFunction) => {
            new BillController(req.body.user).createBill(res, req);
        });
    }
    public billID!: ObjectId;
 
    public writeToBill(res: Response, req: Request, billId: ObjectId, purchase: Purchase) {
        DbClient.push("bills", billId, {
            Purchases: purchase,
        })
            .then((results: any) => {
                res.json({status: "Successfully wrote to bill"});
            })
            .catch((err: any) => {
                res.json({status: "Unable to write to this user's bill on this group"});
            });
    }
 
    public chooseBill(res: Response, req: Request, groupBills: any, purchase: Purchase) {
        for (const groupBill of groupBills) {
            DbClient.getBill(groupBill)
                .then((results: any) => {
                    const userId = results[0].Payee.toString();
                    if (userId === req.body.user) {
                        this.writeToBill(res, req, results[0]._id, purchase);
                    }
                })
                .catch((err: any) => {
                    res.json({status: "Unable to find a bill for this user in this group"});
                });
        }
    }
 
    // Attached to the /purchases/add route and it added a purchase object to a requested bill
    public addPurchase(res: Response, req: Request) {
        const groupID = req.body.groupName;
 
        // find group using group ID, then find specific bill using submitting user
        const purchase = new Purchase(req.body["bill-item-id"],
            req.body["bill-purchase-quantity"],
            req.body["bill-purchase-frequency"]);
 
        // Get group -> pull bills from group
        DbClient.getGroup(groupID)
            .then((results: any) => {
                const groupBills = results[0].Bills;
                this.chooseBill(res, req, groupBills, purchase);
            })
            .catch((err: any) => {
                res.json({status: "Unable to find group"});
            });
    }
 
    public billBelongsToUser(res: Response, req: Request, billId: any) {
        DbClient.getBill(billId)
            .then((results: any) => {
                if (results[0].Payee.toString() === req.query.user) {
                    res.json(results[0]);
                }
            })
            .catch((err: any) => {
                res.json({status: "Cannot find any bills that belong to the current user"});
            });
    }
 
    public parseBillsFromGroup(res: Response, req: Request, userGroups: any) {
        // for(let group of userGroups) {
            DbClient.getGroup(userGroups[1])
                .then((results: any) => {
                    this.billBelongsToUser(res, req, results[0].Bills[0]);
                })
                .catch((err: any) => {
                    res.json({status: "Cannot find the group"});
                });
        // }
    }
 
    public getPurchases(res: Response, req: Request) {
 
        const userId = new ObjectId(req.query.user);
 
        DbClient.getUser2(userId)
            .then((results: any) => {
                this.parseBillsFromGroup(res, req, results[0].Groups);
            })
            .catch((err: any) => {
                res.json({status: "Unable to add purchase to bill"});
            });
    }
 
    // Attached to the /purchases/remove adds a purchase object to the requested bill
    public removePurchase(res: Response, req: Request) {
        DbClient.pull("bills", new ObjectId(req.body.groupName), "Purchases", new ObjectId(this.purchases[0].item.name))
            .then((results: any) => {
                res.redirect("/purchases/view?user=" + req.query.user);
            })
            .catch((err: any) => {
                res.json({status: "Unable to remove purchase from bill"});
            });
    }
 
    // Attached to the /bill/create route and adds a bill to the database
    public createBill(res: Response, req: Request) {
        const payee = req.body.billPayee;
        const payer = req.body.billPayer;
        const groupName = req.body.billGroup;
        if (payee === undefined || payer === undefined || groupName === undefined) {
            // TODO: figure out what page this is going to direct to
            return;
        }
 
        DbClient.insert("bills", {
            Group: groupName,
            Payee: new ObjectId(payee),
            Payer: payer,
            Purchases: [],
        })
            .then((results: any) => {
                res.json(results);
            }).catch((err: any) => {
            res.json({status: "Unable to create bill"});
        });
 
    }
}
